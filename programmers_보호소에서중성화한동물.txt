SELECT i.ANIMAL_ID, i.ANIMAL_TYPE, i.NAME
FROM ANIMAL_INS AS i
JOIN (SELECT ANIMAL_ID FROM ANIMAL_OUTS WHERE SEX_UPON_OUTCOME LIKE '%spayed%' OR SEX_UPON_OUTCOME LIKE '%neutered%') AS o
WHERE i.SEX_UPON_INTAKE NOT LIKE '%spayed%' AND i.SEX_UPON_INTAKE NOT LIKE '%neutered%' AND i.ANIMAL_ID = o.ANIMAL_ID;
